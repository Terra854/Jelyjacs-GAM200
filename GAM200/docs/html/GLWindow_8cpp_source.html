194 void GLWindow::update_time(double fps_calc_interval) {
195  // get elapsed time (in seconds) between previous and current frames
196  static double prev_time = glfwGetTime();
197  double curr_time = glfwGetTime();
198  delta_time = curr_time - prev_time;
199  prev_time = curr_time;
200 
201  // fps calculations
202  static double count = 0.0; // number of game loop iterations
203  static double start_time = glfwGetTime();
204 
205  // get elapsed time since very beginning (in seconds) ...
206  double elapsed_time = curr_time - start_time;
207  if (fps_control) {
208  // sleep for the remaining time if frame was rendered too fast
209  std::this_thread::sleep_for(std::chrono::milliseconds((int)((time_per_frame-delta_time) * 1000)));
210  }
211  ++count;
212 
213  // update fps at least every 10 seconds ...
214  fps_calc_interval = (fps_calc_interval < 0.0) ? 0.0 : fps_calc_interval;
215  fps_calc_interval = (fps_calc_interval > 10.0) ? 10.0 : fps_calc_interval;
216  if (elapsed_time > fps_calc_interval) {
217  GLWindow::fps = count / elapsed_time;
218  start_time = curr_time;
219  count = 0.0;
220  }
221 }
